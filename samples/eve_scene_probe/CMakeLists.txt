function(add_eve_scene_probe_dds_conversion input output metadata)
    add_custom_command(
        OUTPUT ${output} ${metadata}
        COMMAND $<TARGET_FILE:TrinityDdsToRgba>
            --input ${input}
            --output ${output}
            --metadata ${metadata}
            --summary
        DEPENDS TrinityDdsToRgba ${input}
        VERBATIM
    )
endfunction()

                --instance-data
                --manifest ${manifest}
                --summary
            DEPENDS TrinitySharedCacheToGltf ${EVE_SCENE_PROBE_SHARED_CACHE_INDEX}
            VERBATIM
        )
        get_filename_component(cmf_dir ${cmf} DIRECTORY)
        add_custom_command(
            OUTPUT ${cmf}
            COMMAND ${CMAKE_COMMAND} -E make_directory ${cmf_dir}
            COMMAND $<TARGET_FILE:TrinityGltfToCmf>
                --input ${gltf}
                --output ${cmf}
                --instance-data
                --summary
            DEPENDS TrinityGltfToCmf ${gltf} ${bin}
            VERBATIM
        )
    endfunction()
